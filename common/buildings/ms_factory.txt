building_ms_factory_1 = {
	base_buildtime = 360

	category = manufacturing

	potential = {
		exists = owner
		owner = {
			OR = {
				is_gestalt = no
				has_valid_civic = civic_machine_servitor
			}
		}
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = slave_colony }
		owner = {
			OR = {
				has_technology = "tech_BEAMC_1"
				has_technology = "tech_EFSF_BEAMC_1"
			}
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		modifier = {
			job_mobile_suit_engineer_add = 2
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 400
		}
		upkeep = {
			energy = 4
		}
	}

	upgrades = {
		building_ms_factory_2
	}

	ai_weight = {
		weight = 2
		modifier =
		{
			factor = 5
			owner = {
				has_monthly_income = {
					resource = consumer_goods
					value < 5
				}
				has_monthly_income = {
					resource = minerals
					value > 6
				}
			}
		}
	}
}
